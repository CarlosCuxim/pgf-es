\OMIT

\newbox\PGFTESTBOX

\def\BEGINPGFTEST#1{%
  \BEGINTEST{pgf: #1}%
  \setbox\PGFTESTBOX=\hbox\bgroup
    \pgfpicture
}

\def\ENDPGFTEST{%
    \endpgfpicture
  \egroup
  \showbox\PGFTESTBOX
  \ENDTEST
}

\def\BEGINTIKZTEST#1{%
  \BEGINTEST{tikz: #1}%
  \setbox\PGFTESTBOX=\hbox\bgroup
    \tikzpicture
}

\def\ENDTIKZTEST{%
    \endtikzpicture
  \egroup
  \showbox\PGFTESTBOX
  \ENDTEST
}
